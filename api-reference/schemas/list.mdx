## Overview

This endpoint returns a paginated list of all schemas associated with your account. You can filter results and control pagination using query parameters.

<ParamField query="page" type="integer" default="1">
  The page number for pagination (1-indexed)
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Number of schemas to return per page (max 100)
</ParamField>

<ParamField query="search" type="string">
  Search schemas by name or description
</ParamField>

<ParamField query="status" type="string">
  Filter by schema status: `active`, `draft`, `archived`
</ParamField>

<ParamField query="created_after" type="string">
  ISO 8601 timestamp to filter schemas created after this date
</ParamField>

## Response

<ResponseField name="data" type="array">
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      Unique identifier for the schema
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Display name of the schema
    </ResponseField>
    
    <ResponseField name="description" type="string">
      Optional description of the schema
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Current status: `active`, `draft`, or `archived`
    </ResponseField>
    
    <ResponseField name="fields" type="array">
      <Expandable title="field properties">
        <ResponseField name="key" type="string">
          Unique field identifier
        </ResponseField>
        
        <ResponseField name="label" type="string">
          Display label for the field
        </ResponseField>
        
        <ResponseField name="type" type="string">
          Field type: `string`, `number`, `email`, `date`, etc.
        </ResponseField>
        
        <ResponseField name="required" type="boolean">
          Whether the field is required
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp when schema was created
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      ISO 8601 timestamp when schema was last updated
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
  <Expandable title="pagination properties">

    </ResponseField>
    
    <ResponseField name="limit" type="integer">

    </ResponseField>
    
    <ResponseField name="total" type="integer">

    </ResponseField>
    
    <ResponseField name="pages" type="integer">
      Total number of pages
    </ResponseField>
  </Expandable>
</ResponseField>

## Request Examples

<CodeGroup>
```bash cURL

  -H "Authorization: Bearer YOUR_API_KEY" \
  -d "page=1" \
  -d "limit=10" \
  -d "status=active"
```

## Usage Examples

<Tabs>
  <Tab title="Basic Listing">
    ```javascript
    // Get the first page of schemas
    const response = await fetch('https://app.dromo.io/api/v1/schemas', {
      headers: {
        'Authorization': 'Bearer YOUR_API_KEY'
      }
    });
    
    const { data: schemas } = await response.json();
    console.log(`Found ${schemas.length} schemas`);
    ```
  </Tab>
  
  <Tab title="Search & Filter">
    ```javascript
    // Search for active customer-related schemas
    const params = new URLSearchParams({
      search: 'customer',
      status: 'active',
      limit: '50'
    });
    
    const response = await fetch(`https://app.dromo.io/api/v1/schemas?${params}`, {
      headers: {
        'Authorization': 'Bearer YOUR_API_KEY'
      }
    });
    
    const { data: schemas } = await response.json();
    ```
  </Tab>
  
  <Tab title="Pagination">
    ```javascript
    // Fetch all schemas across multiple pages
    let allSchemas = [];
    let page = 1;
    let hasMore = true;
    
    while (hasMore) {
      const response = await fetch(`https://app.dromo.io/api/v1/schemas?page=${page}&limit=100`, {
        headers: {
          'Authorization': 'Bearer YOUR_API_KEY'
        }
      });
      
      const { data: schemas, pagination } = await response.json();
      allSchemas.push(...schemas);
      
      hasMore = page < pagination.pages;
      page++;
    }
    
    console.log(`Total schemas: ${allSchemas.length}`);
    ```
  </Tab>
</Tabs>

<Note>
  **Performance Tip**: Use the `limit` parameter to control response size. For large datasets, consider implementing client-side pagination rather than fetching all records at once.
</Note>
